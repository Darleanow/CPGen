96inline size_t visualWidth(const std::string &s) {
97 std::string stripped;
98 stripped.reserve(s.size());
99
100 bool in_escape = false;
101 for (char c : s) {
102 if (in_escape) {
103 if (std::isalpha(c))
104 in_escape = false;
105 } else if (c == '\033') {
106 in_escape = true;
107 } else {
108 stripped += c;
109 }
110 }

127inline Defs::Key readKey() {
128 char buf[3] = {};
129 read(STDIN_FILENO, buf, 1);
130
131 if (buf[0] == '\r' || buf[0] == '\n')
132 return Defs::Special::Enter;
133 if (buf[0] == 27) {
134 // Wait briefly to see if more bytes follow (escape sequence)
135 // Arrow keys etc. send ESC [ X in quick succession;
136 // a bare Escape press sends only ESC with no follow-up.
137 if (!stdinReady(50))
138 return Defs::Special::Escape; // Bare escape — return immediately
139
140 char seq[2] = {};
141 read(STDIN_FILENO, seq, 1);
142
143 if (seq[0] == '[') {
144 if (stdinReady(50)) {
145 read(STDIN_FILENO, seq + 1, 1);
146 static const std::unordered_map<char, Defs::Special> arrows = {
147 {'A', Defs::Special::Up},
148 {'B', Defs::Special::Down},
149 {'C', Defs::Special::Right},
150 {'D', Defs::Special::Left}};
151
152 if (auto it = arrows.find(seq[1]); it != arrows.end())
153 return it->second;
154 }
155 }
156
157 return Defs::Special::Escape;
158 }
159 if (buf[0] == 127 || buf[0] == 8)
160 return Defs::Special::Backspace;
161 if (buf[0] == '\t')
162 return Defs::Special::Tab;
163
164 return buf[0]; // Regular character
165}
