153inline std::string stripAnsi(const std::string &s) {
154 std::string result;
155 result.reserve(s.size());
156 bool in_escape = false;
157 for (const char c : s) {
158 if (in_escape && std::isalpha(c) != 0) {
159 in_escape = false;
160 continue;
161 }
162 if (in_escape) {
163 continue;
164 }
165 if (c == '\033') {
166 in_escape = true;
167 continue;
168 }
169 result += c;
170 }
171 return result;
172}
