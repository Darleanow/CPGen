158inline std::string stripAnsi(const std::string &s) {
159 std::string result;
160 result.reserve(s.size());
161 bool in_escape = false;
162 for (const char c : s) {
163 if (in_escape && std::isalpha(c) != 0) {
164 in_escape = false;
165 continue;
166 }
167 if (in_escape) {
168 continue;
169 }
170 if (c == '\033') {
171 in_escape = true;
172 continue;
173 }
174 result += c;
175 }
176 return result;
177}
